# 基礎 Linux 指令
# 格式：NAME = COMMAND

# === 系統資訊 ===
顯示系統資訊 = uname -a
顯示磁碟使用量 = df -h
顯示記憶體使用量 = free -h
顯示目前目錄檔案 = ls -la
列出根目錄檔案 = ls -la /
列出使用者目錄檔案 = ls -la /mnt/usr
網路介面資訊 = ifconfig
網路連線狀態 = netstat -an
路由表資訊 = route -n

# === 硬體資訊 ===
核心版本 = cat /proc/version
CPU資訊 = cat /proc/cpuinfo
記憶體資訊 = cat /proc/meminfo
系統運行時間 = uptime
系統時間 = date
硬體時鐘 = hwclock
掛載點 = mount
已載入模組 = lsmod
USB設備 = lsusb
PCI設備 = lspci
I2C匯流排_0 = i2cdetect -y 0
I2C匯流排_1 = i2cdetect -y 1

# === 程序管理 ===
檢查錄影程序 = ps aux | grep hd_video
檢查診斷程序 = ps aux | grep diag
停止錄影程序 = killall hd_video_record_with_vsp_4dev_smart2_pega_dre

# === 檔案管理 ===
列出_mnt_usr_所有檔案 = ls -la /mnt/usr/
列出_var_vsp_所有檔案 = ls -la /var/vsp/*
列出_tmp_tar_所有檔案 = ls -la /tmp/tar
刪除_mnt_usr_JPG檔 = rm -f /mnt/usr/*.jpg
刪除_mnt_usr_YUV檔 = rm -f /mnt/usr/*.yuv
刪除_var_vsp_JPG檔 = rm -f /var/vsp/*.jpg
刪除_var_vsp_YUV檔 = rm -f /var/vsp/*.yuv

# === 系統日誌 ===
最近核心訊息 = dmesg | tail -20
錯誤訊息過濾 = dmesg | grep -i error
攝影機訊息過濾 = dmesg | grep -i camera
系統負載 = cat /proc/loadavg

# === 空間檢查 ===
檢查_tmp空間 = df -h /tmp
檢查_var_vsp空間 = df -h /var/vsp
檢查_mnt_usr空間 = df -h /mnt/usr

# === 系統控制 ===
複製校準資料 = cp -prf /usr/vsp/* /var/vsp/
系統重啟 = reboot
